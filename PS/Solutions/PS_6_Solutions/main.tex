\section*{Global conventions used throughout}
For a level curve $F(x,y)=c$, the gradient $\grad F(x_0,y_0)$ is perpendicular to the tangent line at $(x_0,y_0)$, provided $\grad F(x_0,y_0)\ne 0$. Therefore the tangent line is
\[
F_x(x_0,y_0)(x-x_0)+F_y(x_0,y_0)(y-y_0)=0.
\]
For a level surface $F(x,y,z)=c$, the gradient $\grad F(x_0,y_0,z_0)$ is perpendicular to the tangent plane, so the tangent plane is
\[
F_x(P)(x-x_0)+F_y(P)(y-y_0)+F_z(P)(z-z_0)=0,
\]
where $P=(x_0,y_0,z_0)$.

\section*{Problem 1: Level curves and tangent lines}
\method{Write the curve as $F(x,y)=k$ or $F(x,y)=0$. Compute $\grad F$. At the specified point $P=(x_0,y_0)$, use $\grad F(P)$ as the normal vector to the tangent line.}

\subsection*{1(i) $x^2+y^2=k^2$ at $(k,0)$}
Let
\[
F(x,y)=x^2+y^2.
\]
The level curve is $F(x,y)=k^2$. Its gradient is
\[
\grad F(x,y)=(2x,2y).
\]
At $(k,0)$,
\[
\grad F(k,0)=(2k,0).
\]
Thus the tangent line satisfies
\[
2k(x-k)+0(y-0)=0.
\]
If $k\ne 0$, divide by $2k$:
\[
x-k=0.
\]
\answer{$\boxed{x=k}$. If $k=0$, the curve collapses to the single point $(0,0)$, so there is no ordinary tangent line to a regular curve.}

\subsection*{1(ii) $xy=k$ at $(1,k)$}
Let
\[
F(x,y)=xy.
\]
Then
\[
\grad F(x,y)=(y,x).
\]
At $(1,k)$,
\[
\grad F(1,k)=(k,1).
\]
Therefore the tangent line is
\[
k(x-1)+1(y-k)=0.
\]
Equivalently,
\[
y-k=-k(x-1).
\]
\answer{$\boxed{k(x-1)+y-k=0}$, or $\boxed{y=-kx+2k}$.}

\subsection*{1(iii) $y-x^2=k$ at $(1,k+1)$}
Let
\[
F(x,y)=y-x^2.
\]
Then
\[
\grad F(x,y)=(-2x,1).
\]
At $(1,k+1)$,
\[
\grad F(1,k+1)=(-2,1).
\]
The tangent line is
\[
-2(x-1)+1(y-(k+1))=0.
\]
So
\[
y-k-1=2(x-1),
\]
and therefore
\[
y=2x+k-1.
\]
\answer{$\boxed{y=2x+k-1}$.}

\subsection*{2(i) $z=x^2+y^2$ at $(1,1,2)$}
Here $z=f(x,y)=x^2+y^2$. First check the point:
\[
1^2+1^2=2,
\]
so $(1,1,2)$ lies on the surface. Compute
\[
f_x(x,y)=2x,\qquad f_y(x,y)=2y.
\]
At $(1,1)$,
\[
f_x(1,1)=2,\qquad f_y(1,1)=2.
\]
Therefore
\[
z-2=2(x-1)+2(y-1).
\]
Simplifying,
\[
z=2x+2y-2.
\]
\answer{$\boxed{z=2x+2y-2}$.}

\subsection*{2(ii) $y=x^2-z^2$ at $(4,7,2)$}
First check whether the point lies on the surface. The right-hand side at $(x,z)=(4,2)$ is
\[
4^2-2^2=16-4=12.
\]
But the given $y$-coordinate is $7$, not $12$. Hence
\[
7\ne 12.
\]
\answer{The point $\boxed{(4,7,2)}$ is not on the surface $y=x^2-z^2$. Therefore the requested tangent plane at that point is not defined.}
\warning{If the intended point was $(4,12,2)$, then $y-12=8(x-4)-4(z-2)$, i.e. $y=8x-4z-12$.}

\subsection*{2(iii) $xy+yz+zx=5$ at $(1,2,1)$}
Let
\[
F(x,y,z)=xy+yz+zx-5.
\]
Check the point:
\[
1\cdot 2+2\cdot 1+1\cdot 1=2+2+1=5.
\]
So the point lies on the surface. Compute
\[
F_x=y+z,\qquad F_y=x+z,\qquad F_z=x+y.
\]
At $(1,2,1)$,
\[
\grad F(1,2,1)=(3,2,3).
\]
Thus the tangent plane is
\[
3(x-1)+2(y-2)+3(z-1)=0.
\]
Simplifying,
\[
3x+2y+3z=10.
\]
\answer{$\boxed{3x+2y+3z=10}$.}

\subsection*{2(iv) $\dfrac{x^2}{a^2}+\dfrac{y^2}{b^2}+\dfrac{z^2}{c^2}=1$ at $(x_0,y_0,z_0)$}
Let
\[
F(x,y,z)=\frac{x^2}{a^2}+\frac{y^2}{b^2}+\frac{z^2}{c^2}-1.
\]
Then
\[
\grad F(x,y,z)=\left(\frac{2x}{a^2},\frac{2y}{b^2},\frac{2z}{c^2}\right).
\]
At $(x_0,y_0,z_0)$,
\[
\grad F(x_0,y_0,z_0)=\left(\frac{2x_0}{a^2},\frac{2y_0}{b^2},\frac{2z_0}{c^2}\right).
\]
The tangent plane is
\[
\frac{2x_0}{a^2}(x-x_0)+\frac{2y_0}{b^2}(y-y_0)+\frac{2z_0}{c^2}(z-z_0)=0.
\]
Dividing by $2$,
\[
\frac{x_0x}{a^2}+\frac{y_0y}{b^2}+\frac{z_0z}{c^2}
=\frac{x_0^2}{a^2}+\frac{y_0^2}{b^2}+\frac{z_0^2}{c^2}.
\]
Since $(x_0,y_0,z_0)$ lies on the ellipsoid, the right-hand side is $1$. Hence
\[
\frac{x_0x}{a^2}+\frac{y_0y}{b^2}+\frac{z_0z}{c^2}=1.
\]
\answer{$\boxed{\dfrac{x_0x}{a^2}+\dfrac{y_0y}{b^2}+\dfrac{z_0z}{c^2}=1}$.}

\section*{Problem 3: Chain rule}
\method{When $f=f(x(t),y(t))$, use $\dfrac{df}{dt}=f_x\dfrac{dx}{dt}+f_y\dfrac{dy}{dt}$. When $f=f(x(t),y(t),z(t))$, use $\dfrac{df}{dt}=f_xx'(t)+f_yy'(t)+f_zz'(t)$. The computation is always: first compute partial derivatives, then compute derivatives of the inner functions, then substitute.}

\begin{center}
\begin{tikzpicture}[node distance=1.2cm,>=Latex]
\node[draw,rounded corners,fill=softBlue,minimum width=2.0cm] (t) {$t$};
\node[draw,rounded corners,fill=softGray,right=1.6cm of t] (xy) {$(x(t),y(t),z(t))$};
\node[draw,rounded corners,fill=softGreen,right=1.8cm of xy] (f) {$f(x,y,z)$};
\draw[->,thick] (t)--(xy) node[midway,above] {inner derivatives};
\draw[->,thick] (xy)--(f) node[midway,above] {gradient};
\node[below=0.65cm of xy] {$\displaystyle \frac{df}{dt}=\grad f\cdot \frac{d}{dt}(x,y,z)$};
\end{tikzpicture}
\end{center}

\subsection*{3(i) $f(x,y)=\dfrac{x^3+y^3}{3}$, $x=4\cos t$, $y=3\sin t$}
First compute partial derivatives:
\[
f_x=x^2,\qquad f_y=y^2.
\]
Compute the inner derivatives:
\[
x'(t)=-4\sin t,\qquad y'(t)=3\cos t.
\]
Therefore
\[
\frac{df}{dt}=x^2(-4\sin t)+y^2(3\cos t).
\]
Substitute $x=4\cos t$ and $y=3\sin t$:
\[
\frac{df}{dt}=(4\cos t)^2(-4\sin t)+(3\sin t)^2(3\cos t).
\]
Thus
\[
\frac{df}{dt}=-64\cos^2t\sin t+27\sin^2t\cos t.
\]
\answer{$\boxed{\dfrac{df}{dt}=-64\cos^2t\sin t+27\sin^2t\cos t}$.}

\subsection*{3(ii) $f(x,y)=x^2+y^2+xy$, $x=\sin t$, $y=e^t$}
The partial derivatives are
\[
f_x=2x+y,\qquad f_y=2y+x.
\]
The inner derivatives are
\[
x'(t)=\cos t,\qquad y'(t)=e^t.
\]
Hence
\[
\frac{df}{dt}=(2x+y)x'(t)+(2y+x)y'(t).
\]
Substitute:
\[
\frac{df}{dt}=(2\sin t+e^t)\cos t+(2e^t+\sin t)e^t.
\]
\answer{$\boxed{\dfrac{df}{dt}=(2\sin t+e^t)\cos t+(2e^t+\sin t)e^t}$.}

\subsection*{3(iii) $f(x,y)=\sqrt{1+x^2+y^2}$, $x=\ln t$, $y=\cos t$}
The expression $\ln t$ requires $t>0$. Compute partial derivatives:
\[
f_x=\frac{x}{\sqrt{1+x^2+y^2}},\qquad f_y=\frac{y}{\sqrt{1+x^2+y^2}}.
\]
The inner derivatives are
\[
x'(t)=\frac1t,\qquad y'(t)=-\sin t.
\]
Therefore
\[
\frac{df}{dt}=\frac{x}{\sqrt{1+x^2+y^2}}\cdot\frac1t+\frac{y}{\sqrt{1+x^2+y^2}}(-\sin t).
\]
Substitute $x=\ln t$, $y=\cos t$:
\[
\frac{df}{dt}=\frac{\frac{\ln t}{t}-\cos t\sin t}{\sqrt{1+(\ln t)^2+\cos^2t}}.
\]
\answer{$\boxed{\dfrac{df}{dt}=\dfrac{\frac{\ln t}{t}-\sin t\cos t}{\sqrt{1+(\ln t)^2+\cos^2t}}}$, for $t>0$.}

\subsection*{3(iv) $f(x,y,z)=\ln\sqrt{x^2+y^2+z^2}$, $x=\sin t$, $y=\cos t$, $z=\tan t$}
Write
\[
f(x,y,z)=\frac12\ln(x^2+y^2+z^2).
\]
Therefore
\[
f_x=\frac{x}{x^2+y^2+z^2},\quad f_y=\frac{y}{x^2+y^2+z^2},\quad f_z=\frac{z}{x^2+y^2+z^2}.
\]
Also,
\[
x'(t)=\cos t,\qquad y'(t)=-\sin t,\qquad z'(t)=\sec^2t.
\]
Thus
\[
\frac{df}{dt}=\frac{x\cos t-y\sin t+z\sec^2t}{x^2+y^2+z^2}.
\]
Substitute $x=\sin t$, $y=\cos t$, $z=\tan t$. The numerator becomes
\[
\sin t\cos t-\cos t\sin t+\tan t\sec^2t=\tan t\sec^2t.
\]
The denominator is
\[
\sin^2t+\cos^2t+\tan^2t=1+\tan^2t=\sec^2t.
\]
Therefore
\[
\frac{df}{dt}=\frac{\tan t\sec^2t}{\sec^2t}=\tan t.
\]
\answer{$\boxed{\dfrac{df}{dt}=\tan t}$, wherever all expressions are defined.}

\section*{Problem 4: Implicit differentiation}
\method{For one implicit equation $F(x,y)=0$, if $F_y\ne 0$, then $\dfrac{dy}{dx}=-\dfrac{F_x}{F_y}$. For one implicit equation $F(x,y,z)=0$ defining $z=z(x,y)$, if $F_z\ne 0$, then $\dfrac{\partial z}{\partial x}=-\dfrac{F_x}{F_z}$ and $\dfrac{\partial z}{\partial y}=-\dfrac{F_y}{F_z}$.}

\subsection*{4(i)(a) $x^2-5xy+2y^2=1$}
Let
\[
F(x,y)=x^2-5xy+2y^2-1.
\]
Then
\[
F_x=2x-5y,\qquad F_y=-5x+4y.
\]
Therefore
\[
\frac{dy}{dx}=-\frac{F_x}{F_y}
=-\frac{2x-5y}{-5x+4y}
=\frac{2x-5y}{5x-4y}.
\]
\answer{$\boxed{\dfrac{dy}{dx}=\dfrac{2x-5y}{5x-4y}}$, where $5x-4y\ne 0$.}

\subsection*{4(i)(b) $x^3+y^3-6xy=0$}
Let
\[
F(x,y)=x^3+y^3-6xy.
\]
Then
\[
F_x=3x^2-6y,
\qquad
F_y=3y^2-6x.
\]
Hence
\[
\frac{dy}{dx}=-\frac{3x^2-6y}{3y^2-6x}
=\frac{2y-x^2}{y^2-2x}.
\]
\answer{$\boxed{\dfrac{dy}{dx}=\dfrac{2y-x^2}{y^2-2x}}$, where $y^2-2x\ne 0$.}

\subsection*{4(i)(c) $y\cos x=x^2$}
Let
\[
F(x,y)=y\cos x-x^2.
\]
Then
\[
F_x=-y\sin x-2x,
\qquad
F_y=\cos x.
\]
Therefore
\[
\frac{dy}{dx}=-\frac{-y\sin x-2x}{\cos x}
=\frac{y\sin x+2x}{\cos x}.
\]
\answer{$\boxed{\dfrac{dy}{dx}=\dfrac{y\sin x+2x}{\cos x}}$, where $\cos x\ne 0$.}

\subsection*{4(i)(d) $y^2=\dfrac{\cos(xy)}{1+\sin y}$}
The printed problem is naturally read as the fraction form above. To avoid quotient algebra, first multiply both sides by $1+\sin y$:
\[
y^2(1+\sin y)=\cos(xy).
\]
Define
\[
F(x,y)=y^2(1+\sin y)-\cos(xy).
\]
Then
\[
F_x=y\sin(xy),
\]
because $\partial[-\cos(xy)]/\partial x=y\sin(xy)$. Also,
\[
F_y=2y(1+\sin y)+y^2\cos y+x\sin(xy).
\]
Therefore
\[
\frac{dy}{dx}=-\frac{F_x}{F_y}
=-\frac{y\sin(xy)}{2y(1+\sin y)+y^2\cos y+x\sin(xy)}.
\]
\answer{$\boxed{\dfrac{dy}{dx}=-\dfrac{y\sin(xy)}{2y(1+\sin y)+y^2\cos y+x\sin(xy)}}$.}

\subsection*{4(ii)(a) $x^2+2y^2+3z^2=1$}
Let
\[
F(x,y,z)=x^2+2y^2+3z^2-1.
\]
Then
\[
F_x=2x,\qquad F_y=4y,\qquad F_z=6z.
\]
Hence
\[
z_x=-\frac{2x}{6z}=-\frac{x}{3z},
\qquad
z_y=-\frac{4y}{6z}=-\frac{2y}{3z}.
\]
\answer{$\boxed{z_x=-\dfrac{x}{3z}}$, $\boxed{z_y=-\dfrac{2y}{3z}}$, where $z\ne0$.}

\subsection*{4(ii)(b) $x^2-y^2+z^2-2z=4$}
Let
\[
F(x,y,z)=x^2-y^2+z^2-2z-4.
\]
Then
\[
F_x=2x,\qquad F_y=-2y,\qquad F_z=2z-2.
\]
Thus
\[
z_x=-\frac{2x}{2z-2}=-\frac{x}{z-1},
\qquad
z_y=-\frac{-2y}{2z-2}=\frac{y}{z-1}.
\]
\answer{$\boxed{z_x=-\dfrac{x}{z-1}}$, $\boxed{z_y=\dfrac{y}{z-1}}$, where $z\ne1$.}

\subsection*{4(ii)(c) $x^3+y^3+z^3+4xyz=1$}
Let
\[
F(x,y,z)=x^3+y^3+z^3+4xyz-1.
\]
Then
\[
F_x=3x^2+4yz,
\qquad
F_y=3y^2+4xz,
\qquad
F_z=3z^2+4xy.
\]
Therefore
\[
z_x=-\frac{3x^2+4yz}{3z^2+4xy},
\qquad
z_y=-\frac{3y^2+4xz}{3z^2+4xy}.
\]
\answer{$\boxed{z_x=-\dfrac{3x^2+4yz}{3z^2+4xy}}$, $\boxed{z_y=-\dfrac{3y^2+4xz}{3z^2+4xy}}$.}

\subsection*{4(ii)(d) $yz+x\ln y=z^2$}
Let
\[
F(x,y,z)=yz+x\ln y-z^2.
\]
Then
\[
F_x=\ln y,
\qquad
F_y=z+\frac{x}{y},
\qquad
F_z=y-2z.
\]
Hence
\[
z_x=-\frac{\ln y}{y-2z},
\qquad
z_y=-\frac{z+x/y}{y-2z}.
\]
\answer{$\boxed{z_x=-\dfrac{\ln y}{y-2z}}$, $\boxed{z_y=-\dfrac{z+x/y}{y-2z}}$, where $y>0$ and $y\ne2z$.}

\section*{Problem 5: Mean value theorem}

\subsection*{5(i) Convexity of $S=\{x:\norm{x-x_0}<r\}\subset\R^n$}
\method{To prove a set is convex, take any two points in the set and show that every point on the straight segment joining them is still in the set.}
Let $a,b\in S$ and let $\lambda\in[0,1]$. We must show that
\[
\lambda a+(1-\lambda)b\in S.
\]
By definition of $S$,
\[
\norm{a-x_0}<r,
\qquad
\norm{b-x_0}<r.
\]
Now compute the distance from the convex combination to $x_0$:
\[
\lambda a+(1-\lambda)b-x_0
=\lambda(a-x_0)+(1-\lambda)(b-x_0).
\]
Using the triangle inequality and homogeneity of the norm,
\[
\begin{aligned}
\norm{\lambda a+(1-\lambda)b-x_0}
&=\norm{\lambda(a-x_0)+(1-\lambda)(b-x_0)}\\
&\le \lambda\norm{a-x_0}+(1-\lambda)\norm{b-x_0}\\
&<\lambda r+(1-\lambda)r\\
&=r.
\end{aligned}
\]
Therefore $\lambda a+(1-\lambda)b\in S$.
\answer{$\boxed{S\text{ is convex.}}$}

\subsection*{5(ii) If $D_uf(x)=0$ in every unit direction on an open convex set, then $f$ is constant}
Let $a,b\in S$. If $a=b$, then clearly $f(a)=f(b)$. Assume $a\ne b$.
Define
\[
u=\frac{b-a}{\norm{b-a}}.
\]
Then $u$ is a unit vector. Because $S$ is convex, every point of the segment
\[
\gamma(t)=a+t(b-a),\qquad 0\le t\le1,
\]
lies in $S$. Define a one-variable function
\[
g(t)=f(\gamma(t))=f(a+t(b-a)).
\]
The derivative of $g$ is the rate of change of $f$ along the direction $u$, scaled by the speed $\norm{b-a}$:
\[
g'(t)=\norm{b-a}\,D_uf(\gamma(t)).
\]
By hypothesis, $D_uf(x)=0$ for every $x\in S$ and every unit vector $u$. Therefore
\[
g'(t)=0
\]
for every $t\in[0,1]$ where the derivative is taken. By the one-variable mean value theorem, $g$ is constant on $[0,1]$. Hence
\[
g(0)=g(1).
\]
But
\[
g(0)=f(a),\qquad g(1)=f(b).
\]
Therefore
\[
f(a)=f(b).
\]
Since $a,b\in S$ were arbitrary, $f$ is constant on $S$.
\answer{$\boxed{f\text{ is constant on }S.}$}

\subsection*{5(iii) For differentiable $f$ on open convex $S$, prove $\grad f=0$ iff $f$ is constant}
First suppose $f$ is constant on $S$. Then for each coordinate direction $e_i$,
\[
\frac{\partial f}{\partial x_i}(x)=0
\]
for all $x\in S$. Therefore
\[
\grad f(x)=0
\]
for all $x\in S$.

Conversely, suppose
\[
\grad f(x)=0
\]
for all $x\in S$. For any unit vector $u$,
\[
D_uf(x)=\grad f(x)\cdot u=0\cdot u=0.
\]
By part (ii), $f$ is constant on $S$.
\answer{$\boxed{\grad f=0\text{ on }S\iff f\text{ is constant on }S.}$}

\section*{Problem 6: Linear maps}
\method{A linear map is determined by what it does to basis vectors. In coordinates, every linear map is represented by a matrix. Geometric transformations such as rotations, reflections, and projections are therefore written as matrices acting on coordinate vectors.}

\subsection*{6(i) Rotation in $\R^2$ anticlockwise by angle $\theta$}
The standard anticlockwise rotation matrix is
\[
R_\theta=
\begin{bmatrix}
\cos\theta&-\sin\theta\\
\sin\theta&\cos\theta
\end{bmatrix}.
\]
Therefore
\[
R_\theta\begin{bmatrix}x\\y\end{bmatrix}
=
\begin{bmatrix}
x\cos\theta-y\sin\theta\\
x\sin\theta+y\cos\theta
\end{bmatrix}.
\]
\answer{$\boxed{R_\theta=\begin{bmatrix}\cos\theta&-\sin\theta\\ \sin\theta&\cos\theta\end{bmatrix}}$.}

\draw[->] (0,-0.2)--(0,2.4) node[above] {$y$};
\draw[thick,ashokaBlue,->] (0,0)--(2.1,0.6) node[right] {$v$};
\draw[thick,orange!80!black,->] (0,0)--(1.25,1.8) node[above] {$R_\theta v$};

\node at (0.95,0.7) {$\theta$};
\end{tikzpicture}
\end{center}

\subsection*{6(ii) Reflection about a line making angle $\theta$ with the $x$-axis}
Reflection across the $x$-axis is
\[
\begin{bmatrix}1&0\\0&-1\end{bmatrix}.
\]
To reflect across a line making angle $\theta$, rotate the line to the $x$-axis, reflect, then rotate back:
\[
M_\theta=R_\theta
\begin{bmatrix}1&0\\0&-1\end{bmatrix}
R_{-\theta}.
\]
Carrying out the multiplication gives
\[
M_\theta=
\begin{bmatrix}
\cos 2\theta&\sin 2\theta\\
\sin 2\theta&-\cos 2\theta
\end{bmatrix}.
\]
\answer{$\boxed{M_\theta=\begin{bmatrix}\cos2\theta&\sin2\theta\\ \sin2\theta&-\cos2\theta\end{bmatrix}}$.}

\subsection*{6(iii) Projection onto a line through the origin}
Let the line be spanned by a nonzero vector $v\in\R^n$. The projection of $x$ onto the line is the component of $x$ in the direction of $v$:
\[
\operatorname{proj}_v(x)=\frac{x\cdot v}{v\cdot v}v.
\]
If $u$ is the unit vector in the direction of the line, then $v\cdot v=1$ and
\[
\operatorname{proj}_u(x)=(x\cdot u)u.
\]
In matrix form,
\[
P=\frac{vv^T}{v^Tv}.
\]
For a line in $\R^2$ making angle $\theta$ with the $x$-axis, take $u=(\cos\theta,\sin\theta)$. Then
\[
P_\theta=uu^T=
\begin{bmatrix}
\cos^2\theta&\sin\theta\cos\theta\\
\sin\theta\cos\theta&\sin^2\theta
\end{bmatrix}.
\]
\answer{$\boxed{\operatorname{proj}_v(x)=\dfrac{x\cdot v}{v\cdot v}v}$ and $\boxed{P=\dfrac{vv^T}{v^Tv}}$.}

\draw[->] (0,-1.6)--(0,2.2) node[above] {$y$};

\subsection*{6(iv) Map from $\R^3$ onto the $xy$-plane}
The $xy$-plane consists of all points whose third coordinate is $0$. Therefore the projection map is
\[
P(x,y,z)=(x,y,0).
\]
In matrix form,
\[
P=
\begin{bmatrix}
1&0&0\\
0&1&0\\
0&0&0
\end{bmatrix}.
\]
\answer{$\boxed{P(x,y,z)=(x,y,0)}$ and $\boxed{P=\begin{bmatrix}1&0&0\\0&1&0\\0&0&0\end{bmatrix}}$.}

\subsection*{6(v) Every linear map $A:\R^n\to\R^m$ has an $m\times n$ matrix}
Let $e_1,\dots,e_n$ be the standard basis of $\R^n$. Every vector $x\in\R^n$ can be written as
\[
x=x_1e_1+x_2e_2+\cdots+x_ne_n.
\]
Since $A$ is linear,
\[
A(x)=x_1A(e_1)+x_2A(e_2)+\cdots+x_nA(e_n).
\]
Now form the matrix $\widetilde A$ whose $j$th column is $A(e_j)$:
\[
\widetilde A=\begin{bmatrix} A(e_1)&A(e_2)&\cdots&A(e_n)\end{bmatrix}.
\]
Then matrix multiplication gives exactly
\[
A(x)=\widetilde A
\begin{bmatrix}
x_1\\x_2\\\vdots\\x_n
\end{bmatrix}.
\]
Conversely, any $m\times n$ matrix defines a linear map by multiplication because matrix multiplication preserves sums and scalar multiples.
\answer{$\boxed{\text{Linear maps }\R^n\to\R^m\text{ are exactly }m\times n\text{ matrices.}}$}

\section*{Problems 7 and 8: Second-degree Taylor polynomials and Hessians at critical points}
\method{First solve $\grad f=0$ to find the critical points. Then compute the Hessian matrix
\[
H_f(a,b)=\begin{bmatrix}f_{xx}(a,b)&f_{xy}(a,b)\\f_{yx}(a,b)&f_{yy}(a,b)\end{bmatrix}.
\]
At a critical point $(a,b)$, the second-degree Taylor polynomial is
\[
T_2(x,y)=f(a,b)+\frac12
\begin{bmatrix}x-a&y-b\end{bmatrix}
H_f(a,b)
\begin{bmatrix}x-a\\y-b\end{bmatrix},
\]
because the first-order term is zero when $\grad f(a,b)=0$.}

\begin{center}
\begin{tikzpicture}[scale=1.0,>=Latex]
\node[draw,rounded corners,fill=softBlue,minimum width=3.6cm] (g) {Solve $\grad f=0$};
\node[draw,rounded corners,fill=softGray,right=1.4cm of g,minimum width=3.6cm] (h) {Compute Hessian $H_f$};
\node[draw,rounded corners,fill=softGreen,right=1.4cm of h,minimum width=3.8cm] (t) {Insert into $T_2$};
\draw[->,thick] (g)--(h);
\draw[->,thick] (h)--(t);
\end{tikzpicture}
\end{center}

\subsection*{7(i), 8(i) $f(x,y)=xy^2+x^2y$}
Compute the gradient:
\[
f_x=y^2+2xy=y(y+2x),\qquad f_y=2xy+x^2=x(2y+x).
\]
Set both equal to zero:
\[
y(y+2x)=0,
\qquad
x(2y+x)=0.
\]
Checking the cases gives only
\[
(x,y)=(0,0).
\]
The Hessian entries are
\[
f_{xx}=2y,
\qquad
f_{xy}=2x+2y,
\qquad
f_{yy}=2x.
\]
At $(0,0)$,
\[
H_f(0,0)=\begin{bmatrix}0&0\\0&0\end{bmatrix}.
\]
Also $f(0,0)=0$. Therefore
\[
T_2(x,y)=0.
\]
\answer{Critical point: $\boxed{(0,0)}$. Hessian: $\boxed{\begin{bmatrix}0&0\\0&0\end{bmatrix}}$. Taylor polynomial: $\boxed{T_2=0}$.}

\subsection*{7(ii), 8(ii) $f(x,y)=x^2+(y-1)^2$}
\[
f_x=2x,
\qquad
f_y=2(y-1).
\]
Thus $\grad f=0$ gives
\[
x=0,
\qquad y=1.
\]
So the critical point is $(0,1)$. The Hessian is constant:
\[
H_f=\begin{bmatrix}2&0\\0&2\end{bmatrix}.
\]
Since $f(0,1)=0$,
\[
T_2(x,y)=x^2+(y-1)^2.
\]
\answer{Critical point: $\boxed{(0,1)}$. Hessian: $\boxed{\begin{bmatrix}2&0\\0&2\end{bmatrix}}$. Taylor polynomial: $\boxed{T_2=x^2+(y-1)^2}$.}

\subsection*{7(iii), 8(iii) $f(x,y)=1+x^2-(y-1)^2$}
\[
f_x=2x,
\qquad
f_y=-2(y-1).
\]
Thus $\grad f=0$ gives $(0,1)$. The Hessian is
\[
H_f=\begin{bmatrix}2&0\\0&-2\end{bmatrix}.
\]
Since $f(0,1)=1$,
\[
T_2(x,y)=1+x^2-(y-1)^2.
\]
\answer{Critical point: $\boxed{(0,1)}$. Hessian: $\boxed{\begin{bmatrix}2&0\\0&-2\end{bmatrix}}$. Taylor polynomial: $\boxed{T_2=1+x^2-(y-1)^2}$.}

\subsection*{7(iv), 8(iv) $f(x,y)=(x-y+1)^2$}
Let $q=x-y+1$. Then
\[
f_x=2q,
\qquad
f_y=-2q.
\]
Thus $\grad f=0$ exactly when
\[
q=0,
\qquad\text{that is}\qquad y=x+1.
\]
So every point $(a,a+1)$ is critical. The Hessian is constant:
\[
H_f=\begin{bmatrix}2&-2\\-2&2\end{bmatrix}.
\]
At a critical point $(a,a+1)$, $f(a,a+1)=0$. Let
\[
u=x-a,
\qquad
v=y-(a+1).
\]
Then
\[
T_2(x,y)=\frac12\begin{bmatrix}u&v\end{bmatrix}
\begin{bmatrix}2&-2\\-2&2\end{bmatrix}
\begin{bmatrix}u\\v\end{bmatrix}
=(u-v)^2.
\]
Since $u-v=(x-a)-(y-a-1)=x-y+1$,
\[
T_2(x,y)=(x-y+1)^2.
\]
\answer{Critical set: $\boxed{y=x+1}$. Hessian: $\boxed{\begin{bmatrix}2&-2\\-2&2\end{bmatrix}}$. Taylor polynomial at $(a,a+1)$: $\boxed{T_2=(x-y+1)^2}$.}

\subsection*{7(v), 8(v) $f(x,y)=\sin x\sin y\sin(x+y)$ on $[0,\pi]^2$}
First compute the first partial derivatives:
\[
\begin{aligned}
f_x
&=\cos x\sin y\sin(x+y)+\sin x\sin y\cos(x+y)\\
&=\sin y\sin(2x+y),
\end{aligned}
\]
and similarly
\[
f_y=\sin x\sin(x+2y).
\]
Thus $\grad f=0$ means
\[
\sin y\sin(2x+y)=0,
\qquad
\sin x\sin(x+2y)=0.
\]
On the boundary, this gives the four corners
\[
(0,0),\quad (\pi,0),\quad (0,\pi),\quad (\pi,\pi).
\]
In the interior, $\sin x\ne0$ and $\sin y\ne0$, so
\[
\sin(2x+y)=0,
\qquad
\sin(x+2y)=0.
\]
Since $x,y\in(0,\pi)$, the only interior solutions are
\[
(x,y)=\left(\frac\pi3,\frac\pi3\right),
\qquad
(x,y)=\left(\frac{2\pi}{3},\frac{2\pi}{3}\right).
\]
Now compute the second derivatives:
\[
f_{xx}=2\sin y\cos(2x+y),
\qquad
f_{xy}=\sin(2x+2y),
\qquad
f_{yy}=2\sin x\cos(x+2y).
\]
At the four corners, $H_f=0$ and $f=0$, so $T_2=0$.

At $p_1=(\pi/3,\pi/3)$,
\[
f(p_1)=\frac{3\sqrt3}{8},
\qquad
H_f(p_1)=
\begin{bmatrix}
-\sqrt3&-\sqrt3/2\\
-\sqrt3/2&-\sqrt3
\end{bmatrix}.
\]
Let $u=x-\pi/3$ and $v=y-\pi/3$. Then
\[
T_2=\frac{3\sqrt3}{8}-\frac{\sqrt3}{2}u^2-\frac{\sqrt3}{2}uv-\frac{\sqrt3}{2}v^2.
\]
At $p_2=(2\pi/3,2\pi/3)$,
\[
f(p_2)=-\frac{3\sqrt3}{8},
\qquad
H_f(p_2)=
\begin{bmatrix}
\sqrt3&\sqrt3/2\\
\sqrt3/2&\sqrt3
\end{bmatrix}.
\]
Let $u=x-2\pi/3$ and $v=y-2\pi/3$. Then
\[
T_2=-\frac{3\sqrt3}{8}+\frac{\sqrt3}{2}u^2+\frac{\sqrt3}{2}uv+\frac{\sqrt3}{2}v^2.
\]
\answer{Critical points are $\boxed{(0,0),(\pi,0),(0,\pi),(\pi,\pi),(\pi/3,\pi/3),(2\pi/3,2\pi/3)}$. The Hessians and Taylor polynomials are listed above explicitly.}

\subsection*{7(vi), 8(vi) $f(x,y)=x^2-xy+y^2-2x+y$}
\[
f_x=2x-y-2,
\qquad
f_y=-x+2y+1.
\]
Set both equal to zero:
\[
2x-y-2=0,
\qquad
-x+2y+1=0.
\]
From the first equation $y=2x-2$. Substitute into the second:
\[
-x+2(2x-2)+1=0
\quad\Longrightarrow\quad
3x-3=0.
\]
Thus $x=1$ and $y=0$. The Hessian is constant:
\[
H_f=\begin{bmatrix}2&-1\\-1&2\end{bmatrix}.
\]
Also $f(1,0)=-1$. Let $u=x-1$ and $v=y$. Then
\[
T_2=-1+u^2-uv+v^2.
\]
\answer{Critical point: $\boxed{(1,0)}$. Hessian: $\boxed{\begin{bmatrix}2&-1\\-1&2\end{bmatrix}}$. Taylor polynomial: $\boxed{T_2=-1+(x-1)^2-(x-1)y+y^2}$.}

\subsection*{7(vii), 8(vii) $f(x,y)=2x^2-xy-3y^2-3x+7y$}
\[
f_x=4x-y-3,
\qquad
f_y=-x-6y+7.
\]
Set both equal to zero:
\[
4x-y-3=0,
\qquad
-x-6y+7=0.
\]
The first gives $y=4x-3$. Substitute:
\[
-x-6(4x-3)+7=0
\quad\Longrightarrow\quad
-25x+25=0.
\]
So $x=1$, and then $y=1$. The Hessian is
\[
H_f=\begin{bmatrix}4&-1\\-1&-6\end{bmatrix}.
\]
Also $f(1,1)=2$. Let $u=x-1$, $v=y-1$. Then
\[
T_2=2+2u^2-uv-3v^2.
\]
\answer{Critical point: $\boxed{(1,1)}$. Hessian: $\boxed{\begin{bmatrix}4&-1\\-1&-6\end{bmatrix}}$. Taylor polynomial: $\boxed{T_2=2+2(x-1)^2-(x-1)(y-1)-3(y-1)^2}$.}

\subsection*{7(viii), 8(viii) $f(x,y)=x^3-3xy^2+y^3$}
\[
f_x=3x^2-3y^2=3(x-y)(x+y),
\qquad
f_y=-6xy+3y^2=3y(y-2x).
\]
Set both equal to zero. From $f_y=0$, either $y=0$ or $y=2x$. If $y=0$, then $f_x=3x^2$, so $x=0$. If $y=2x$, then
\[
f_x=3x^2-3(2x)^2=-9x^2,
\]
so again $x=0$ and $y=0$. Thus the only critical point is $(0,0)$.

The second derivatives are
\[
f_{xx}=6x,
\qquad
f_{xy}=-6y,
\qquad
f_{yy}=-6x+6y.
\]
At $(0,0)$,
\[
H_f(0,0)=\begin{bmatrix}0&0\\0&0\end{bmatrix}.
\]
Since $f(0,0)=0$,
\[
T_2=0.
\]
\answer{Critical point: $\boxed{(0,0)}$. Hessian: $\boxed{0}$. Taylor polynomial: $\boxed{T_2=0}$.}

\subsection*{7(ix), 8(ix) $f(x,y)=x^2y^3(6-x-y)$}
First compute
\[
f_x=x y^3(12-3x-2y),
\qquad
f_y=x^2y^2(18-3x-4y).
\]
Thus $\grad f=0$ if either $x=0$, or $y=0$, or both linear factors vanish:
\[
12-3x-2y=0,
\qquad
18-3x-4y=0.
\]
Solving these two equations gives
\[
y=3,
\qquad x=2.
\]
Therefore the critical set is
\[
\{(0,b):b\in\R\}\cup\{(a,0):a\in\R\}\cup\{(2,3)\}.
\]
The second derivatives are
\[
\begin{aligned}
f_{xx}&=y^3(12-6x-2y),\\
f_{xy}&=xy^2(36-9x-8y),\\
f_{yy}&=x^2y(36-6x-12y).
\end{aligned}
\]
At $(0,b)$,
\[
H_f(0,b)=\begin{bmatrix}b^3(12-2b)&0\\0&0\end{bmatrix}.
\]
Since $f(0,b)=0$, the Taylor polynomial at $(0,b)$ is
\[
T_2=b^3(6-b)x^2.
\]
At $(a,0)$,
\[
H_f(a,0)=\begin{bmatrix}0&0\\0&0\end{bmatrix},
\qquad
T_2=0.
\]
At $(2,3)$,
\[
f(2,3)=2^2\cdot 3^3(6-2-3)=108,
\]
and
\[
H_f(2,3)=\begin{bmatrix}-162&-108\\-108&-144\end{bmatrix}.
\]
Let $u=x-2$, $v=y-3$. Then
\[
T_2=108-81u^2-108uv-72v^2.
\]
\answer{Critical set: $\boxed{x=0\text{ or }y=0\text{ or }(x,y)=(2,3)}$. Hessians and Taylor polynomials are as computed above.}

\subsection*{7(x), 8(x) $f(x,y)=e^{2x+3y}(8x^2-6xy+3y^2)$}
Let
\[
q(x,y)=8x^2-6xy+3y^2,
\qquad
f=e^{2x+3y}q.
\]
Then
\[
f_x=e^{2x+3y}(2q+q_x),
\qquad
f_y=e^{2x+3y}(3q+q_y),
\]
where
\[
q_x=16x-6y,
\qquad
q_y=-6x+6y.
\]
Because $e^{2x+3y}\ne0$, the critical point equations are
\[
2q+16x-6y=0,
\qquad
3q-6x+6y=0.
\]
From the second equation,
\[
q=2x-2y.
\]
From the first equation,
\[
q=-8x+3y.
\]
Equating these two expressions for $q$ gives
\[
2x-2y=-8x+3y
\quad\Longrightarrow\quad
10x=5y
\quad\Longrightarrow\quad
y=2x.
\]
Substitute $y=2x$ into $q=2x-2y$:
\[
q=2x-4x=-2x.
\]
But directly from $q=8x^2-6x(2x)+3(2x)^2$,
\[
q=8x^2-12x^2+12x^2=8x^2.
\]
Thus
\[
8x^2=-2x
\quad\Longrightarrow\quad
2x(4x+1)=0.
\]
Hence
\[
x=0\quad\text{or}\quad x=-\frac14.
\]
Using $y=2x$, the critical points are
\[
(0,0),
\qquad
\left(-\frac14,-\frac12\right).
\]
At a critical point the Hessian simplifies to the derivative of the bracketed first-derivative factors. The resulting Hessian values are
\[
H_f(0,0)=\begin{bmatrix}16&-6\\-6&6\end{bmatrix},
\]
and
\[
H_f\left(-\frac14,-\frac12\right)
=e^{-2}\begin{bmatrix}14&-9\\-9&3/2\end{bmatrix}.
\]
At $(0,0)$, $f(0,0)=0$, so
\[
T_2=8x^2-6xy+3y^2.
\]
At $p=(-1/4,-1/2)$,
\[
f(p)=e^{-2}\cdot \frac12=\frac{e^{-2}}2.
\]
Let
\[
u=x+\frac14,
\qquad
v=y+\frac12.
\]
Then
\[
T_2=\frac{e^{-2}}2+e^{-2}\left(7u^2-9uv+\frac34v^2\right).
\]
\answer{Critical points: $\boxed{(0,0)}$ and $\boxed{(-1/4,-1/2)}$. Hessians and Taylor polynomials are as above.}

\section*{Problem 9: First and second degree Taylor polynomial of $f(x,y)=e^x\cos y$ at the origin}
\method{For Taylor expansion at $(0,0)$, compute $f(0,0)$, the gradient at $(0,0)$, and the Hessian at $(0,0)$. Then use
\[
T_1=f(0,0)+f_x(0,0)x+f_y(0,0)y,
\]
and
\[
T_2=T_1+\frac12\left(f_{xx}(0,0)x^2+2f_{xy}(0,0)xy+f_{yy}(0,0)y^2\right).
\]}

First,
\[
f(0,0)=e^0\cos0=1.
\]
The first partial derivatives are
\[
f_x=e^x\cos y,
\qquad
f_y=-e^x\sin y.
\]
At $(0,0)$,
\[
f_x(0,0)=1,
\qquad
f_y(0,0)=0.
\]
Therefore the first-degree Taylor polynomial is
\[
T_1(x,y)=1+x.
\]
Now compute second partial derivatives:
\[
f_{xx}=e^x\cos y,
\qquad
f_{xy}=-e^x\sin y,
\qquad
f_{yy}=-e^x\cos y.
\]
At $(0,0)$,
\[
f_{xx}(0,0)=1,
\qquad
f_{xy}(0,0)=0,
\qquad
f_{yy}(0,0)=-1.
\]
Thus
\[
T_2(x,y)=1+x+\frac12(x^2-y^2).
\]
\answer{$\boxed{T_1(x,y)=1+x}$ and $\boxed{T_2(x,y)=1+x+\dfrac12(x^2-y^2)}$.}
